class Solution {
public:
    int numJewelsInStones(string J, string S) {
        
        int cnt = 0;
        
            for(int j=0;j<S.length();j++)
            {
                for(int i=0;i<J.length();j++)
                {
                    if(S[j]==J[i])
                        cnt++;
                }
            }
            
               
        
        return cnt;
        
    }
};
